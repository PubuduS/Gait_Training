262 private float GenerateNormalRandom( float mean, float sigma, int min, int max )
263 {
264 float rand1 = UnityEngine.Random.Range(0.0f, 1.0f);
265 float rand2 = UnityEngine.Random.Range(0.0f, 1.0f);
266
267 float n = Mathf.Sqrt( -2.0f * Mathf.Log(rand1)) * Mathf.Cos( ( 2.0f * Mathf.PI ) * rand2 );
268
269 float generatedNumber = mean + sigma * n;
270
271 generatedNumber = Mathf.Clamp(generatedNumber, min, max);
272
273 return generatedNumber;
274 }
